Phase 2: Step 2 — The Visual Admin Controller
We are staying in your Next.js Admin Panel. We are going to upgrade that EncounterSubTab so you can pick your background and see the battle layout.

1. Add Background Selection
You need to be able to pick an "Arena" (like a Seoul street or a dungeon) from your Supabase storage.

2. Add the Battle Preview
This is a small window that sits next to your form. It shows:

Left Side: Your player sprite.

Right Side: The monster sprite you just selected.

Background: The arena image you chose.

The "God Mode" Admin Code Update
Replace your previous EncounterSubTab with this version. It adds the Background URL to your metadata and includes the Live Preview.

TypeScript
const EncounterSubTab = ({ activeMapId, supabase }) => {
  const [name, setName] = useState('');
  const [selectedMonster, setSelectedMonster] = useState('');
  const [selectedBg, setSelectedBg] = useState(''); // New: For the arena
  
  // Rewards matching your DB columns
  const [exp, setExp] = useState(0);
  const [coins, setCoins] = useState(0);
  const [gems, setGems] = useState(0);

  const handleSave = async () => {
    const automatedMetadata = {
      rewards: { exp, coins, gems },
      visuals: {
        bg_url: selectedBg,      // Background choice
        monster_url: selectedMonster,
        layout: 'SIDE_VIEW'      // Tells React Native: Player Left / Monster Right
      }
    };

    const { error } = await supabase.from('encounter_pool').insert([{
      map_id: activeMapId,
      name: name,
      icon_url: selectedMonster,
      metadata: automatedMetadata 
    }]);

    if (!error) alert("Battle Encounter Live!");
  };

  return (
    <div className="flex flex-row gap-8 p-6 bg-black">
      {/* LEFT: The Form */}
      <div className="w-1/2 space-y-4">
        <h3 className="text-cyan-400 font-black uppercase text-sm">Battle Config</h3>
        <input placeholder="Enemy Name" onChange={(e) => setName(e.target.value)} className="w-full bg-gray-900 border border-gray-800 p-2 text-white" />
        
        <label className="text-[10px] text-gray-500 uppercase font-bold">Pick Background</label>
        {/* You'll plug your IconGallery here */}
        <div className="h-20 bg-gray-900 rounded border border-gray-800" /> 

        <button onClick={handleSave} className="w-full bg-cyan-600 p-3 font-black uppercase">Save Encounter</button>
      </div>

      {/* RIGHT: The Pokémon-Style Preview */}
      <div className="w-1/2 border-2 border-cyan-500/30 rounded-lg relative overflow-hidden aspect-video bg-slate-900">
        {selectedBg && <img src={selectedBg} className="absolute inset-0 w-full h-full object-cover opacity-50" />}
        
        {/* Player Sprite (Bottom Left) */}
        <div className="absolute bottom-4 left-4 w-24 h-24 bg-cyan-500/20 rounded-full blur-xl" />
        <img src="PLAYER_SPRITE_URL" className="absolute bottom-6 left-8 w-20 h-20 object-contain" />

        {/* Monster Sprite (Top Right) */}
        <img src={selectedMonster} className="absolute top-6 right-8 w-20 h-20 object-contain" />
        <p className="absolute top-2 right-8 text-cyan-400 font-bold text-[10px] uppercase">{name}</p>
      </div>
    </div>
  );
};
