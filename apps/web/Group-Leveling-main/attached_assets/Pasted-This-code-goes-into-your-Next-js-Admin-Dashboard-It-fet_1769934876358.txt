This code goes into your Next.js Admin Dashboard. It fetches everything from your world-assets bucket and displays it in a clean, selectable grid.

TypeScript
const IconGallery = ({ onSelect, selectedUrl }) => {
  const [icons, setIcons] = useState([]);
  const [loading, setLoading] = useState(true);

  const loadGallery = async () => {
    setLoading(true);
    // Fetch all files from the 'icons' folder
    const { data, error } = await supabase.storage
      .from('world-assets')
      .list('icons');

    if (data) {
      const iconUrls = data.map(file => ({
        name: file.name,
        url: supabase.storage.from('world-assets').getPublicUrl(`icons/${file.name}`).data.publicUrl
      }));
      setIcons(iconUrls);
    }
    setLoading(false);
  };

  useEffect(() => { loadGallery(); }, []);

  return (
    <div className="space-y-2">
      <div className="flex justify-between items-center">
        <label className="text-[10px] font-black uppercase text-gray-500">Quick-Select Icons</label>
        <button onClick={loadGallery} className="text-[10px] text-cyan-500 hover:underline">Refresh</button>
      </div>
      
      <div className="grid grid-cols-5 gap-2 p-2 bg-black/50 border border-gray-800 rounded-lg max-h-40 overflow-y-auto">
        {icons.map((icon) => (
          <button
            key={icon.name}
            onClick={() => onSelect(icon.url)}
            className={`aspect-square p-1 rounded transition-all ${
              selectedUrl === icon.url ? 'bg-cyan-500/20 border border-cyan-500' : 'bg-transparent border border-transparent'
            }`}
          >
            <img src={icon.url} className="w-full h-full object-contain" alt={icon.name} />
          </button>
        ))}
      </div>
    </div>
  );
};
