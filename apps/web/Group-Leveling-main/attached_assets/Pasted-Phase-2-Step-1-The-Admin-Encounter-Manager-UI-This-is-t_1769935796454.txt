Phase 2: Step 1 — The Admin "Encounter Manager" UI
This is the sub-tab inside your World Map Admin. This is where you’ll actually "plop" your Monsters and Pets into the RNG pool.

Here is the plan to build this screen right now:

1. The Layout
We will split your sidebar into two toggles: "World Plotter" (for icons on the map) and "Encounter Pool" (for the RNG monsters/pets).

2. The Form (The "No-JSON" Builder)
This form will have the input boxes for Gold, XP, and HP that you mentioned. It will also include the Icon Gallery so you can pick the monster's picture.

The Code for your Next.js Sub-Tab
Copy this into your Admin Panel components. This handles the "No-JSON" automation you liked:

TypeScript
import { useState } from 'react';

const EncounterSubTab = ({ activeMapId, supabase }) => {
  const [eventType, setEventType] = useState('MONSTER');
  const [name, setName] = useState('');
  const [chance, setChance] = useState(0.05); // 5% chance
  const [selectedIcon, setSelectedIcon] = useState('');
  
  // No-JSON Reward States
  const [gold, setGold] = useState(0);
  const [xp, setXp] = useState(0);
  const [hp, setHp] = useState(100);

  const handleSave = async () => {
    // AUTOMATION: This creates the JSON object for you
    const automatedMetadata = {
      reward_gold: gold,
      reward_xp: xp,
      base_hp: hp,
      is_catchable: eventType === 'PET',
      layout: 'POKEMON_STYLE' // Tells the phone to put player left / monster right
    };

    const { error } = await supabase.from('encounter_pool').insert([{
      map_id: activeMapId,
      event_type: eventType,
      name: name,
      spawn_chance: chance,
      icon_url: selectedIcon,
      metadata: automatedMetadata // The JSON is handled!
    }]);

    if (!error) alert(`${name} is now live in the game!`);
  };

  return (
    <div className="flex flex-col gap-4 p-4 bg-gray-900/80 rounded-lg border border-cyan-900/30">
      <h3 className="text-cyan-400 font-black uppercase text-xs">RNG Event Creator</h3>
      
      {/* 1. Event Type */}
      <select 
        value={eventType} 
        onChange={(e) => setEventType(e.target.value)}
        className="bg-black border border-gray-800 p-2 text-xs text-white"
      >
        <option value="MONSTER">Monster (Battle)</option>
        <option value="PET">Wild Pet (Catch)</option>
        <option value="LOOT">Loot Box (Find)</option>
      </select>

      {/* 2. Basic Info */}
      <input 
        placeholder="Encounter Name" 
        onChange={(e) => setName(e.target.value)}
        className="bg-black border border-gray-800 p-2 text-xs text-white"
      />

      {/* 3. The Reward Builder (No-JSON typing!) */}
      <div className="grid grid-cols-3 gap-2">
        <div className="flex flex-col">
          <label className="text-[10px] text-gray-500 uppercase">Gold</label>
          <input type="number" onChange={(e) => setGold(Number(e.target.value))} className="bg-black border border-gray-700 p-1 text-xs" />
        </div>
        <div className="flex flex-col">
          <label className="text-[10px] text-gray-500 uppercase">XP</label>
          <input type="number" onChange={(e) => setXp(Number(e.target.value))} className="bg-black border border-gray-700 p-1 text-xs" />
        </div>
        <div className="flex flex-col">
          <label className="text-[10px] text-gray-500 uppercase">HP</label>
          <input type="number" onChange={(e) => setHp(Number(e.target.value))} className="bg-black border border-gray-700 p-1 text-xs" />
        </div>
      </div>

      <button 
        onClick={handleSave}
        className="bg-cyan-600 hover:bg-cyan-500 text-white font-black py-2 rounded text-xs uppercase"
      >
        Add to RNG Pool
      </button>
    </div>
  );
};
